eoanb_influence_calc = {
	every_country = {
	    clear_variable = pp_scaling_influence
	    if = {
	        limit = { check_variable = { overall_score > 0 } }
		    set_variable = { pp_scaling_influence = overall_score }
	    }
        else = {
		    set_variable = { pp_scaling_influence = 1 }
		}		
	}
}
